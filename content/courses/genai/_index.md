(function(){

      if(!tile) return;

        tile.classList.add('cc-done');
        if(box){ box.classList.add('cc-checked'); box.innerHTML = '&#10003;'; }
        var label = tile.querySelector('.cc-mlabel');
        if(label) label.style.color = '#1b5e20';
      } else {
        tile.classList.remove('cc-done');
        if(box){ box.classList.remove('cc-checked'); box.innerHTML = ''; }
        var label = tile.querySelector('.cc-mlabel');
        if(label) label.style.color = '';
      }
    });
  }

  // When a module tile is clicked, mark it complete in localStorage
  // This fires BEFORE navigation — so when learner returns, the tile is green
  ALL_MODULES.forEach(function(id){
    var tile = document.getElementById(id);
    if(!tile) return;
    tile.addEventListener('click', function(){
      var done = getCompleted();
      if(done.indexOf(id) === -1){
        done.push(id);
        try{ localStorage.setItem('genai_done', JSON.stringify(done)); }
        catch(e){}
      }
    });
  });

  // Run on page load
  updateProgress();
})();
</script>
